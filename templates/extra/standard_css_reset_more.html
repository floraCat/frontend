//以下为其他常用重置样式参考

html,body{ overflow-x:hidden;}      //防止屏幕小于网页宽度出现横向滚动条
body{ position:relative;}      //当用body布局及内部元素定位时可兼容ie6,ie7
iframe{ border:0;}      
input,button,textarea{ background:transparent;}      //此些控件默认带白色或灰色背景，重置成透明背景
table{ border-collapse:collapse; border-spacing:0;}      //table常用设置
